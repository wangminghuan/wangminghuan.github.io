еҸ‘иЎЁдәҺ 2019-11-04 | жӣҙж–°дәҺ 2019-12-05 | еҲҶзұ»дәҺ ES6зі»еҲ— | йҳ…иҜ»ж¬Ўж•°пјҡ

жҰӮиҝ°

Proxy: иӢұж–Үж„ҸжҖқдёәпјҡд»ЈзҗҶпјҢиҜ»йҹідёәпјҡ[ЛҲprЙ’ksi]гҖӮе®ғеңЁзӣ®ж ҮеҜ№иұЎд№ӢеүҚжһ¶и®ҫдёҖеұӮвҖңжӢҰжҲӘвҖқпјҢеӨ–з•ҢеҜ№иҜҘеҜ№иұЎзҡ„и®ҝй—®пјҢйғҪеҝ…йЎ»е…ҲйҖҡиҝҮиҝҷеұӮжӢҰжҲӘпјҢз”ЁдәҺдҝ®ж”№жҹҗдәӣж“ҚдҪңзҡ„й»ҳи®ӨиЎҢдёәпјҢзӯүеҗҢдәҺеңЁиҜӯиЁҖеұӮйқўеҒҡеҮәдҝ®ж”№пјҢжүҖд»ҘеұһдәҺдёҖз§ҚвҖңе…ғзј–зЁӢвҖқпјҲmeta programmingпјүпјҢеҚіеҜ№зј–зЁӢиҜӯиЁҖиҝӣиЎҢзј–зЁӢгҖӮ

ReflectеҜ№иұЎдёҺProxyеҜ№иұЎдёҖж ·пјҢд№ҹжҳҜ ES6 дёәдәҶж“ҚдҪңеҜ№иұЎиҖҢжҸҗдҫӣзҡ„ж–° APIгҖӮ

жҲ‘д»¬е…ҲзңӢдёҖдёӘдҫӢеӯҗ

var obj = new Proxy({}, {
  get(target, key, receiver) {
    console.log(`getting ${key}!`);

  },
  set(target, key, value, receiver) {
    console.log(`setting ${key}!`);
    return Reflect.set(target, key, value, receiver);
  }
});

Proxy еҜ№иұЎ

Proxy жһ„йҖ еҮҪж•°

ES6 еҺҹз”ҹжҸҗдҫӣ Proxy жһ„йҖ еҮҪж•°пјҢз”ЁжқҘз”ҹжҲҗ Proxy е®һдҫӢгҖӮ

var proxy = new Proxy(target, handler);
  • targetеҸӮж•°иЎЁзӨәжүҖиҰҒжӢҰжҲӘзҡ„зӣ®ж ҮеҜ№иұЎ
  • handlerеҸӮж•°д№ҹжҳҜдёҖдёӘеҜ№иұЎпјҢз”ЁжқҘе®ҡеҲ¶жӢҰжҲӘиЎҢдёә

жҲ‘д»¬еңЁtargetеҜ№иұЎдёҠи®ҫзҪ®дёҖеұӮжӢҰжҲӘпјҡ

var target = {};
var handler = {
  get(target, property){
    return 'Li'
  },
};
var proxy = new Proxy(target, handler);

еҪ“дҝ®ж”№targetеҜ№иұЎж—¶

target.name="jack";

еҸҜд»ҘзңӢеҲ°proxyе…¶е®һжҳҜжҺҘж”¶еҲ°еҸҳеҢ–дәҶ

дҪҶеҪ“жҲ‘д»¬и®ҝй—®proxyдёӢеұһжҖ§ж—¶пјҢиҝ”еӣһзҡ„дҫқж—§жҳҜLi:

apply()

applyж–№жі•жӢҰжҲӘеҮҪж•°зҡ„и°ғз”ЁгҖҒcallе’Ңapplyж“ҚдҪң, applyж–№жі•еҸҜд»ҘжҺҘеҸ—дёүдёӘеҸӮж•°:

apply(target, object, args)

- target: зӣ®ж ҮеҜ№иұЎ
- object: зӣ®ж ҮеҜ№иұЎзҡ„дёҠдёӢж–ҮеҜ№иұЎпјҲthisпјү
- args: зӣ®ж ҮеҜ№иұЎзҡ„еҸӮж•°ж•°з»„

жӢҰжҲӘ Proxy е®һдҫӢдҪңдёәеҮҪж•°и°ғз”Ёзҡ„ж“ҚдҪңпјҢжҜ”еҰӮ proxy(вҖҰargs)гҖҒ proxy.call(object, вҖҰargs) гҖҒproxy.apply(вҖҰ)гҖҒReflect.apply(proxy,вҖҰ)йғҪдјҡиў«жӢҰжҲӘгҖӮ

    console.log(target===foo)
    console.log([...args])
    return "I am Proxy apply"
  }
})

has()

hasж–№жі•з”ЁжқҘжӢҰжҲӘHasPropertyж“ҚдҪңпјҢеҚіеҲӨж–ӯеҜ№иұЎжҳҜеҗҰе…·жңүжҹҗдёӘеұһжҖ§ж—¶пјҢиҝҷдёӘж–№жі•дјҡз”ҹж•ҲгҖӮеҸҜд»ҘжҺҘеҸ—дёӨдёӘеҸӮж•°пјҢ

has(target, propKey)

- target: зӣ®ж ҮеҜ№иұЎ
- propKey: йңҖжҹҘиҜўзҡ„еұһжҖ§еҗҚ

жӢҰжҲӘ propKey in proxyзҡ„ж“ҚдҪңпјҢиҝ”еӣһдёҖдёӘеёғе°”еҖјгҖӮе…ёеһӢзҡ„ж“ҚдҪңе°ұжҳҜinиҝҗз®—з¬ҰгҖӮ

var handler = {
  has (target, key) {
    if (key[0] === '_') {
      return false;
    }
    return key in target;
  }
};
var target = { _prop: 'foo', prop: 'bar' };
var proxy = new Proxy(target, handler);
'_prop' in proxy; //false
'prop' in proxy;//true

hasж–№жі•жӢҰжҲӘзҡ„жҳҜHasPropertyж“ҚдҪңпјҢиҖҢдёҚжҳҜHasOwnPropertyж“ҚдҪңпјҢеҚіhasж–№жі•дёҚеҲӨж–ӯдёҖдёӘеұһжҖ§жҳҜеҜ№иұЎиҮӘиә«зҡ„еұһжҖ§пјҢиҝҳжҳҜз»§жүҝзҡ„еұһжҖ§гҖӮ

construct()

construct(target, args)

deleteProperty()

deletePropertyж–№жі•з”ЁдәҺжӢҰжҲӘdeleteж“ҚдҪңпјҢеҰӮжһңиҝҷдёӘж–№жі•жҠӣеҮәй”ҷиҜҜжҲ–иҖ…иҝ”еӣһfalseпјҢеҪ“еүҚеұһжҖ§е°ұж— жі•иў«deleteе‘Ҫд»ӨеҲ йҷӨгҖӮ

deleteProperty(target, propKey)

- targetпјҡзӣ®ж ҮеҜ№иұЎ
- propKeyпјҡиҰҒеҲ йҷӨзҡ„еұһжҖ§еҗҚ

еҲ йҷӨвҖқ_вҖқејҖеӨҙзҡ„еұһжҖ§дјҡжҠҘй”ҷпјҡ

  var handler = {
    deleteProperty (target, propKey) {
    if(propKey.match(/^_/)){

    name:"Tom",

defineProperty()

Object.defineProperty() ж–№жі•дјҡзӣҙжҺҘеңЁдёҖдёӘеҜ№иұЎдёҠе®ҡд№үдёҖдёӘж–°еұһжҖ§пјҢжҲ–иҖ…дҝ®ж”№дёҖдёӘеҜ№иұЎзҡ„зҺ°жңүеұһжҖ§пјҢ е№¶иҝ”еӣһиҝҷдёӘеҜ№иұЎ

definePropertyж–№жі•дјҡжӢҰжҲӘObject.defineProperty е’Ң Object.definePropertiesж“ҚдҪңгҖӮиҝ”еӣһдёҖдёӘеёғе°”еҖј

defineProperty (target, propKey, descriptor)

- targetпјҡзӣ®ж ҮеҜ№иұЎ
- propKeyпјҡиҰҒеҲ йҷӨзҡ„еұһжҖ§еҗҚ
- descriptor: е°Ҷиў«е®ҡд№үжҲ–дҝ®ж”№зҡ„еұһжҖ§зҡ„жҸҸиҝ°з¬ҰпјҢеҸӮи§ҒObject.definePropertyдёӯзҡ„жҸҸиҝ°з¬Ұ

дҫӢеҰӮпјҡ

var handler = {
  defineProperty (target, key, descriptor) {
    return false;
  }
};
var target = {};
var proxy = new Proxy(target, handler);
